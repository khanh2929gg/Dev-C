    C D E F G H I

A B C D E F G H I J K
  B C D E F G H I J
    C D E F G H I

main(){
	int n;scanf("%d",&n);
	int i,j;
	for ( i = 0; i < 2*n - 1; i++) {
        for ( j = 0; j < 2*n - 1; j++) {
            if ((i - n + 1)*(i - n + 1) + (j - n + 1)*(j - n + 1) <= n*n)
                printf("%c ", 'A'+j);
            else printf("  ");
        }
        printf("\n");
    }
}
